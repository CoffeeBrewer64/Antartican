# Dictionary

This is the full length dictionary. It is **NOT** in alphabetical order.

# Greetings

ogi = hi/hello (ogi)

a'ska = bye (uhska)

# Common Words

gee = yes (gee)

ga = no (ga)

# Conjunctions

a'nk = and (uhnk)

# Question words

onti = what (onti)

onoi = who (onoy)

ondi = when (ondee)

onke = where (onke)

onzi = why (onzee)

onbi = how

# Emotions

ek' = sad (ekuh)

el' = happy (eluh)

nag' = angry (naguh)

nimpo = calm
